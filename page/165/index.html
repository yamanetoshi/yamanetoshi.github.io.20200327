(define empty-env
  (lambda ()
    (lambda (sym search)
      (if search
          #f
          (eopl:error 'apply-env "No binding for ~s" sym)))))

(define extend-env
  (lambda (syms valus env)
    (lambda (sym search)
      (let ((pos (list-find-position sym syms)))
        (if (number? pos)
          (if search

            (list-ref vals pos))
          (apply-env env sym))))))

(define apply-env
