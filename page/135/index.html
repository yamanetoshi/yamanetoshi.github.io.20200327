sed 自分メモ

for i in * ;
do
  sed -i -e "/hoge 1 15 525/a hoge 0 0 0" $i
  sed -i -e "s/fuga 0 1 2 3¥$/fuga 0 1 2 3 300/g" $i
  sed -i -e "s/piyo 0 1 2 3¥$/piyo 0 1 2 3 300/g" $i
end
